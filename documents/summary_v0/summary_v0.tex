			\section{Overview}
			\begin{itemize}
				\item Based on TE-830M1-0025 MEMS chip package
				\item Run on STM32 LQFP64 chip-series up to 72MHz PLL clock
				\item Can be interfaced at 115200 Serial protocol
				\item Can operated on low-power batteries
			\end{itemize}
